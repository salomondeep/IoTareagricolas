int sensorPin = A0;
int sensorValue = 0;
int relay = D1;  

void setup() {
  Serial.begin(921600);
  pinMode(relay, OUTPUT);
}

void loop() {
  sensorValue = analogRead(sensorPin);
  Serial.println(sensorValue);


  if(sensorValue != 1024)
  {
    digitalWrite(relay, HIGH);
    }else{
      digitalWrite(relay, LOW);
    }
}
